\section{\textbf{Problem}}
Let  $\mathbb{F}$ be a filed and let $f$ be the linear functional on $\mathbb{F}^2$ defined by,
\begin{align}
f(x_1,x_2)  = ax_1 + bx_2 \label{1.1}
\end{align}
For given linear operator  \textbf{T}, such that
\begin{align}
\textbf{T}(x_1, x_2) = (-x_2, x_1)  \label{1.2}
\end{align}
Let 
\begin{align}
g = \textbf{T}^{t} f
\end{align}
find $g(x_1, x_2)$

\section{solution}

We have,
\begin{align}
g = \textbf{T}^{t}f\\ 
\implies g(x_1, x_2) = \textbf{T}^{t} f(x_1, x_2)\label{2.2}
\end{align}
We know that, if \textbf{V} and \textbf{W} be vector spaces over the field $\mathbb{F}$. For each linear transformation \textbf{T} from \textbf{V} into \textbf{W}, there is a unique linear transformation
$\textbf{T}^{t}$ from $\textbf{W*}$ into $\textbf{V*}$ such that,

\begin{align}
(\textbf{T}^{t}g)(\alpha)  = g (\textbf{T}\alpha) \label{2.3}
\end{align}
Where for every $g$ in $\textbf{W*}$ and $\alpha$ in \textbf{V}.\\

Now using \eqref{2.3} in \eqref{2.2} we can write,
\begin{align}
 \textbf{T}^{t} f(x_1, x_2) = f(\textbf{T}(x_1, x_2)) \label{2.4}\\
\intertext{Using \eqref{1.2} in \eqref{2.4}}
f(\textbf{T}(x_1, x_2)) = f(-x_2, x_1)
\intertext{From \eqref{1.1}, we can write}
f(-x_2, x_1) = -ax_2 + bx_1
\intertext{Hence,}
 \textbf{T}^{t} f(x_1, x_2) = -ax_2 + bx_1
\end{align}
